from tf.keras.layers import Dense
from tf.keras.activations import sigmoid

class ResetGate(tf.keras.layers.Layer):
  def __init__(self, units):
    self.units = units
    self.Wr = Dense(self.units)
    self.W = Dense(self.units)

  def call(self, x, z):
    r_t = sigmoid(self.gate(x))
    y = self.W(tf.math.multiply(r_t, x))
    h_hat = tf.math.tanh(y)
    return tf.math.multiply(h_hat, z)

from tf.keras.layers import Dense
from tf.keras.activations import sigmoid

class UpdateGate(tf.keras.layers.Layer):
  def __init__(self, units):
    self.units = units
    self.Wz = Dense(self.units)

  def call(self, x):
    z_t = sigmoid(self.Wz(x))
    return z_t
class GRU(tf.keras.Model):
  def __init__(self, units):
    self.units = units
    self.update_gate = UpdateGate(self.units)
    self.reset_gate = ResetGate(self.units)

  def call(self, x,h):
    xh = tf.concat([x,h], axis=1)
    z = self.update_gate(xh)
    zh_hat = self.reset_gate(xh, z)
    h_new = tf.math.multiply((1-z), h) + zh_hat
    return h_new
